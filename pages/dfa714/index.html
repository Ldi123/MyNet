# 🚀 动态规划通用解题技巧和步骤

# 📋 标准解题五步法

# 🎈 Step 1: 确定DP含义（最关键！）

核心思想：明确dp[i]或dp[i][j]代表什么

# 常见定义模式：

// 模式1：以i结尾的最优解
dp[i] = 以第i个元素结尾的最大/最小/数量

// 模式2：前i个元素的最优解  
dp[i] = 前i个元素的最大/最小/数量

// 模式3：从起点到(i,j)的最优解
dp[i][j] = 从(0,0)到(i,j)的最优解

// 模式4：区间DP
dp[i][j] = 区间[i,j]内的最优解

// 模式5：状态DP
dp[i][state] = 第i个位置在state状态下的最优解

# 🔄 Step 2: 推导状态转移方程

核心：当前状态如何从之前状态推导而来

# 常见转移模式：

// 选择型：选或不选
dp[i] = max(dp[i-1], dp[i-1] + nums[i])

// 路径型：从哪里来
dp[i][j] = dp[i-1][j] + dp[i][j-1]

// 决策型：多种选择取最优
dp[i] = min(dp[i-1], dp[i-2], dp[i-3]) + cost[i]

// 区间型：分割点枚举
dp[i][j] = min(dp[i][k] + dp[k+1][j]) for k in [i,j-1]

# 🔹 最大子数组和

public int maxSubArray(int[] nums) {
    // dp[i] = 以nums[i]结尾的最大子数组和
    int dp = nums[0], result = nums[0];
    for (int i = 1; i < nums.length; i++) {
        dp = Math.max(nums[i], dp + nums[i]);
        result = Math.max(result, dp);
    }
    return result;
}

# 🔹 最长递增子序列

public int lengthOfLIS(int[] nums) {
    // dp[i] = 以nums[i]结尾的最长递增子序列长度
    int[] dp = new int[nums.length];
    Arrays.fill(dp, 1);
    int result = 1;
    
    for (int i = 1; i < nums.length; i++) {
        for (int j = 0; j < i; j++) {
            if (nums[j] < nums[i]) {
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
        result = Math.max(result, dp[i]);
    }
    return result;
}

# 📍 2. 背包DP

# 🔹 0-1背包模板

public int knapsack01(int[] weights, int[] values, int W) {
    // dp[i][w] = 前i个物品，容量w的最大价值
    int[][] dp = new int[weights.length + 1][W + 1];
    
    for (int i = 1; i <= weights.length; i++) {
        for (int w = 1; w <= W; w++) {
            // 不选第i个物品
            dp[i][w] = dp[i-1][w];
            // 选第i个物品（如果放得下）
            if (w >= weights[i-1]) {
                dp[i][w] = Math.max(dp[i][w], 
                    dp[i-1][w-weights[i-1]] + values[i-1]);
            }
        }
    }
    return dp[weights.length][W];
}

# 🔹 完全背包模板

public int completeKnapsack(int[] weights, int[] values, int W) {
    // dp[w] = 容量w的最大价值
    int[] dp = new int[W + 1];
    
    for (int i = 0; i < weights.length; i++) {
        for (int w = weights[i]; w <= W; w++) {
            dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
        }
    }
    return dp[W];
}

# 📍 3. 区间DP

# 🔹 区间DP模板

public int intervalDP(int[] nums) {
    int n = nums.length;
    // dp[i][j] = 区间[i,j]的最优解
    int[][] dp = new int[n][n];
    
    // 按区间长度递增
    for (int len = 2; len <= n; len++) {
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            dp[i][j] = Integer.MAX_VALUE;
            
            // 枚举分割点k
            for (int k = i; k < j; k++) {
                dp[i][j] = Math.min(dp[i][j], 
                    dp[i][k] + dp[k+1][j] + cost(i, k, j));
            }
        }
    }
    return dp[0][n-1];
}

# 📍 4. 状态机DP

# 🔹 买卖股票模板

public int maxProfit(int[] prices) {
    // 状态：0-持有股票，1-不持有股票
    int hold = -prices[0];  // 持有股票的最大利润
    int sold = 0;           // 不持有股票的最大利润
    
    for (int i = 1; i < prices.length; i++) {
        int newSold = Math.max(sold, hold + prices[i]);  // 卖出
        int newHold = Math.max(hold, sold - prices[i]);  // 买入
        sold = newSold;
        hold = newHold;
    }
    return sold;
}

# 📍 5. 路径DP

# 🔹 网格路径模板

public int uniquePaths(int m, int n) {
    // dp[i][j] = 到达(i,j)的路径数
    int[][] dp = new int[m][n];
    
    // 初始化边界
    for (int i = 0; i < m; i++) dp[i][0] = 1;
    for (int j = 0; j < n; j++) dp[0][j] = 1;
    
    // 状态转移
    for (int i = 1; i < m; i++) {
        for (int j = 1; j < n; j++) {
            dp[i][j] = dp[i-1][j] + dp[i][j-1];
        }
    }
    return dp[m-1][n-1];
}

# 🎉 小结：压缩判断技巧

# ✅ 可以压缩：

  1. 状态转移只看前面有限行/列
  2. 没有"跳跃式"依赖（如dp[i][j]依赖dp[i-3][j-2]）
  3. 转移方程中的索引有规律

# ❌ 难以压缩：

  1. 需要查看任意历史状态
  2. 状态转移有复杂的跳跃模式
  3. 需要同时保存多个维度的历史信息

# 🔧 压缩步骤：

  1. 分析状态转移方程
  2. 确定依赖关系
  3. 选择合适的遍历顺序
  4. 处理边界和中间值保存
  5. 验证正确性

// 错误：不知道dp[i]具体含义

# ❌ 常见错误3: 遍历顺序错误

// 错误：使用了尚未计算的状态
for (int i = 0; i < n; i++) {
    dp[i] = dp[i+1] + nums[i];  // dp[i+1]还没计算！
}

// 正确：保证依赖的状态已计算
for (int i = n-2; i >= 0; i--) {
    dp[i] = dp[i+1] + nums[i];
